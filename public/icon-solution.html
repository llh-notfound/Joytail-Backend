💻 前端集成代码

1. 引入CSS文件

<link rel="stylesheet" href="/css/petpal-icons.css">

2. HTML结构

<div class="petpal-interaction-bar"> <div class="petpal-interaction-left"> <button class="petpal-btn" onclick="handleLike(postId)"> <i class="petpal-icon petpal-heart-outline"></i> <span class="petpal-count">{{likes}}</span> </button> <button class="petpal-btn" onclick="handleCollect(postId)"> <i class="petpal-icon petpal-bookmark-outline"></i> <span class="petpal-count">{{collects}}</span> </button> </div> </div>

3. JavaScript交互逻辑

function handleLike(postId) { fetch(`/api/community/posts/${postId}/like`, { method: 'POST', headers: { 'Authorization': 'Bearer ' + token } }) .then(res => res.json()) .then(data => { if (data.code === 200) { const button = document.querySelector(`[data-post="${postId}"] .like-btn`); if (data.data.isLiked) { button.classList.add('liked'); } else { button.classList.remove('liked'); } button.querySelector('.petpal-count').textContent = data.data.likesCount; } }); } function handleCollect(postId) { fetch(`/api/community/posts/${postId}/collect`, { method: 'POST', headers: { 'Authorization': 'Bearer ' + token } }) .then(res => res.json()) .then(data => { if (data.code === 200) { const button = document.querySelector(`[data-post="${postId}"] .collect-btn`); if (data.data.isCollected) { button.classList.add('collected'); } else { button.classList.remove('collected'); } button.querySelector('.petpal-count').textContent = data.data.collectsCount; } }); }
🔧 使用说明：
  • 方案一：使用Font Awesome图标库（推荐）- 包含丰富的图标和动画效果
  • 方案二：使用自定义SVG图标 - 更轻量，可完全自定义
  • 后端API返回 isLiked 和 isCollected 字段控制图标状态
  • 所有样式和动画效果已包含在CSS文件中
